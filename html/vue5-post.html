JS
    
        var app = new Vue({
            el:"#app", 
            data:{
                attachRed: false,
                color: 'green'
            },
            computed:{
                duvClasses: function(){
                    return{
                        red: this.attachRed,
                        green: !this.attachRed
                    }
                }
            }
        })
    
HTML
    
        <div id="app2" class="demo-box">
            <div class="demo"@click="attachRed = !attachRed":class="{red: attachRed, green: !attachRed}">1 </div>
            <div class="demo":class="color">2 </div>
            <div class="demo":class="divClass">3 <div>
            <div>2번 상자의 색:  <input type="text" v-model="color"> </div>
        </div>
